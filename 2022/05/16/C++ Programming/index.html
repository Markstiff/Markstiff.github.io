6、重载运算符：

Struct vec
{
        int x, y;
        long long len() {…}
        bool operator < {const vec &rhs} const
        {
                return len() <= rhs.len();
        }
        long long operator *(const vec &rhs)  const
        {
                 return 111 * x * rhs.x + 111 * y * rhs.y;
         }
        long long operator ^(const vec &rhs) const
        {
                 return 111 * x * rhs.y - 111 * y * rhs.x;
        }
};
